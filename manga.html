// ================================================================ // WORKER - MANGADEX PROXY DENGAN 3 FALLBACK GAMBAR // ================================================================ const API_BASE = 'https://api.mangadex.org'; async function handleRequest(request) { const url = new URL(request.url); const path = url.pathname; // ---------- PROXY GAMBAR (dengan 3 fallback) ---------- if (path === '/image/') { const imageUrl = url.searchParams.get('url'); if (!imageUrl) { return new Response( JSON.stringify({ error: 'Parameter "url" diperlukan' }), { status: 400, headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*' } } ); } // Coba 3 metode const methods = [ // Metode 1: Langsung ke uploads.mangadex.org async () => { return await fetch(imageUrl, { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36', 'Referer': 'https://mangadex.org/', 'Accept': 'image/webp,image/apng,image/*,*/*;q=0.8' } }); }, // Metode 2: Via corsproxy.io async () => { const proxyUrl = `https://corsproxy.io/?${encodeURIComponent(imageUrl)}`; return await fetch(proxyUrl, { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36' } }); }, // Metode 3: Via allorigins async () => { const proxyUrl = `https://api.allorigins.win/raw?url=${encodeURIComponent(imageUrl)}`; return await fetch(proxyUrl); } ]; let lastError = null; for (let i = 0; i < methods.length; i++) { try { console.log(`[IMAGE] Mencoba metode ${i + 1}...`); const response = await methods[i](); if (response.ok) { console.log(`[IMAGE] ✅ Metode ${i + 1} berhasil`); const newResponse = new Response(response.body, response); newResponse.headers.set('Access-Control-Allow-Origin', '*'); newResponse.headers.set('Cache-Control', 'public, max-age=86400'); return newResponse; } console.log(`[IMAGE] Metode ${i + 1} gagal: ${response.status}`); lastError = `Metode ${i + 1} gagal: ${response.status}`; } catch (e) { console.log(`[IMAGE] Metode ${i + 1} error: ${e.message}`); lastError = `Metode ${i + 1} error: ${e.message}`; } } // Semua metode gagal return new Response( JSON.stringify({ error: 'Semua metode gagal mengambil gambar', detail: lastError, imageUrl: imageUrl }), { status: 500, headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*' } } ); } // ---------- PROXY API ---------- if (path.startsWith('/manga') || path.startsWith('/chapter') || path.startsWith('/at-home') || path.startsWith('/author') || path.startsWith('/cover')) { const targetUrl = `${API_BASE}${path}${url.search}`; try { const newRequest = new Request(targetUrl, { method: request.method, headers: request.headers, body: (request.method !== 'GET' && request.method !== 'HEAD' && request.method !== 'OPTIONS') ? request.body : undefined, }); newRequest.headers.delete('Via'); let response = await fetch(newRequest); if (!response.ok) { throw new Error(`HTTP ${response.status} - ${response.statusText}`); } response = new Response(response.body, response); response.headers.set('Access-Control-Allow-Origin', '*'); response.headers.set('Access-Control-Allow-Methods', 'GET, POST, PUT, DELETE, OPTIONS'); response.headers.set('Access-Control-Allow-Headers', '*'); response.headers.set('Access-Control-Max-Age', '86400'); return response; } catch (error) { return new Response( JSON.stringify({ error: 'Gagal mengakses API MangaDex', message: error.message }), { status: 500, headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*' } } ); } } // ---------- ROOT ---------- if (path === '/' || path === '') { return new Response( JSON.stringify({ status: 'ok', message: 'Worker berjalan dengan 3 fallback image proxy', methods: ['direct', 'corsproxy.io', 'allorigins'] }), { status: 200, headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*' } } ); } return new Response( JSON.stringify({ error: 'Endpoint tidak dikenal', path: path }), { status: 404, headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*' } } ); } export default { async fetch(request) { if (request.method === 'OPTIONS') { return new Response(null, { status: 204, headers: { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE, OPTIONS', 'Access-Control-Allow-Headers': '*', 'Access-Control-Max-Age': '86400' } }); } return handleRequest(request); } };
